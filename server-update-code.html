# Start the server update process
echo ""
echo "Mise à jour commencée à:"
date
sleep 2s


# Get the updated site from the Github, and delete the local old one
echo ""
echo "- MISE A JOUR DES DONNEES SUR LE BUREAU -"
sleep 1s
echo "Mise à jour des données du serveur en cours..."
sleep 0.5s
## Delete the old version of the site
echo ""
echo ">>> Suppression de la copie des données du serveur datée sur le bureau..."
cd ~/Desktop
rm -rf WolfY_D3v_Server_BACKUP_REPOSITORY
echo ">>> Copie datée des données du serveur supprimée avec succès!"
sleep 0.5s
## Fetching the updated version of the site from the Github
echo ""
echo ">>> Clonage du Github du serveur sur le bureau en cours..."
git clone https://WolfYD3v:[MON_TOKEN]@github.com/WolfYD3v/WolfY_D3v_Server_BACKUP_REPOSITORY.git
echo ">>> Github cloné dans le bureau avec succès!"
sleep 0.5s


# Copying the bureau version to the right directory
echo ""
echo "- CHANGEMENT DES DONNEES DU SITE -"
sleep 1s
## Go the the right folder
cd /var/www/html
sleep 0.5s
## Delete the old version stored
echo "psw" | sudo -S -k rm -rf WolfY_D3v_Server_BACKUP_REPOSITORY
sleep 0.5s
## Copy the updated version on the desktop into the right folder
echo "psw" | sudo -S -k cp -r ~/Desktop/WolfY_D3v_Server_BACKUP_REPOSITORY .
sleep 0.5s


# Delete the version in the desktop
cd ~/Desktop
rm -rf WolfY_D3v_Server_BACKUP_REPOSITORY
echo ""


# Restarting Apache2 service
echo ""
echo "- RESTARTING APACHE2 SERVICE -"
echo "psw" | sudo -S -k service apache2 restart
sleep 0.5s


# Update process ended after a 5 seconds delay
cd
echo ""
echo "Mise à jour des données du serveur finie à:"
date
sleep 5s
clear
